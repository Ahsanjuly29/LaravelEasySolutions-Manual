How to Array Unique

// array unique function will make it unique 
                $a = array("a"=>"red","b"=>"green","c"=>"red"); // Example of array
                
                // SORT_STRING - Default. Compare items as strings
                $a = array_unique($a);
            
                // SORT_REGULAR - Compare items normally (don't change types)
                $a = array_unique($a, SORT_REGULAR);

                // SORT_NUMERIC - Compare items numerically
                $a = array_unique($a, SORT_NUMERIC);

                // SORT_LOCALE_STRING - Compare items as strings, based on current locale
                $a = array_unique($a, SORT_LOCALE_STRING);
